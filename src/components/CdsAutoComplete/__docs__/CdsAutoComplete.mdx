# CdsAutoComplete

The `CdsAutoComplete` component is a search-based dropdown list that allows users to select an option from a predefined set.

## Props

- **`label`** (`string`, default: `""`)  
  - The label displayed above the combobox.

- **`options`** (`{ id: string, label: string }[]`, default: `[]`)  
  - The list of available options in the dropdown.

- **`disabled`** (`boolean`, default: `false`)  
  - Disables the dropdown if set to `true`.

- **`size`** (`"small" | "medium" | "large"`, default: `"medium"`)  
  - Sets the size of the dropdown.

- **`onChange`** (`(event: ChangeEvent<HTMLInputElement>) => void`, default: `undefined`)  
  - Callback function triggered when an option is selected.

---

## Usage

### Basic Example

```tsx
import React, { useState } from "react";
import CdsAutoComplete from "../CdsAutoComplete";

const CdsAutoCompleteExample = () => {
  const [selectedValue, setSelectedValue] = useState("");

  const sampleOptions = [
    { id: "1", label: "Option 1" },
    { id: "2", label: "Option 2" },
    { id: "3", label: "Option 3" },
    { id: "4", label: "Option 4" },
    { id: "5", label: "Option 5" },
  ];

  const handleAutoCompleteChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    setSelectedValue(event.target.value);
  };

  return (
    <CdsAutoComplete
      label="Select an option"
      options={sampleOptions}
      size="medium"
      onChange={handleAutoCompleteChange}
    />
  );
};

export default CdsAutoCompleteExample;
